def irb
  t = Plist4r::Table.new
  t.cell 0,0, "foo"
  t.cell 0,0

  t.cell 2,0, "pineapples"
  t.cell 4,0, "some_string"
  t.fill 0..5, 1..4, 0
  puts t.inspect_a
end
