def merge! obj
  case obj
  when Hash
    Attributes.each do |a|
      hash = obj
      if hash[a.to_sym]
        value = hash[a.to_sym].deep_clone
        self.send( (a+'=').to_sym, value)
      end
    end
  when Haml4r::TableCell
    Attributes.each do |a|
      value = args[0].send(a.to_sym).deep_clone
      self.send( (a+'=').to_sym, value)
    end
  else
    raise "unsupported type"
  end
end
