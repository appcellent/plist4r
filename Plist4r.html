def string_detect_format string
  # puts "in string_detect_format"
  # puts "string = #{string.inspect}"
  # s = string.strip
  string.strip! if string[0,1] =~ /\s/
  # s = string
  # puts "s = #{s.inspect}"
  case string[0,1]
  when "{","("
    :next_step
  when "b"
    if string =~ /^bplist/
      :binary
    else
      nil
    end
  when "<"
    if string =~ /^\<\?xml/ && string =~ /\<\!DOCTYPE plist/
      :xml
    else
      nil
    end
  else
    nil
  end
end
