def string_detect_format string
  string.strip! if string[0,1] =~ /\s/
  case string[0,1]
  when "{","("
    :gnustep
  when "b"
    if string =~ /^bplist/
      :binary
    else
      nil
    end
  when "<"
    if string =~ /^\<\?xml/ && string =~ /\<\!DOCTYPE plist/
      :xml
    else
      nil
    end
  else
    nil
  end
end
