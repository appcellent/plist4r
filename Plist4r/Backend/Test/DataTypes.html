def gen_mixed
  @plist.edit do
    [true,false].each do |bool|
      set "BoolKey#{bool.to_s.capitalize}", bool
    end

    (0..9).each do |i|
      set "IntegerKey#{(65+i).chr}", i
    end

    (0..25).each do |i|
      set "StringKey#{(65+i).chr}", "#{(97+i).chr}"*(i+1)
    end

    (0..25).each do |i|
      a = []
      (0..i).each do |j|
        a << "String#{(65+j).chr}"
      end
      set "ArrayKey#{(65+i).chr}", a
    end

    (0..25).each do |i|
      h = Plist4r::OrderedHash.new
      (0..i).each do |j|
        h["String#{(65+j).chr}"] = "#{(97+j).chr}"*(j+1)
      end
      set "HashKey#{(65+i).chr}", h
    end
  end
end

def gen_plist_1024
  @plist_1024 ||= Plist4r.new do
    coarse_multiplier = 18
    (0..coarse_multiplier).each do |i|

      a = []
      (0..25).each do |j|
        a << "String#{(65+j).chr}"
      end
      set "ArrayKey#{i}", a

      h = Plist4r::OrderedHash.new
      (0..25).each do |j|
        h["String#{(65+j).chr}"] = "#{(97+j).chr}"*(j+1)
      end
      set "HashKey#{i}", h
    end
    
    a = []
    (0..11).each do |j|
      a << "String#{(65+j).chr}"
    end
    set "ArrayKeyPad1", a
    
    a = []
    (0..23).each do |j|
      a << "String#{(65+j).chr}"
    end
    set "ArrayKeyPad2", a
  end
  @plist_1024
end
