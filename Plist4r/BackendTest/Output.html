def to_html
  require 'haml'
  engine = ::Haml::Engine.new to_html_haml
  rendered_html_output = engine.render self
  
  docs_dir = File.dirname(__FILE__) + "/../../../lib/plist4r/docs"
  File.open "#{docs_dir}/BackendTestMatrix.html","w" do |o|
    o << rendered_html_output
  end
end
