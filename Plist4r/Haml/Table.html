Overview

A TableView is composed of 3 sub-tables. The table of Column Headers, the table of Row Headers, and the table of table data definitions. Its the responsibility of the TableView to render all Table elements into haml, by providing accessor methods.

The three tables are laid out by TableView as follows: 

  |--------------------------------|
  |                |-------------| |
  |                |             | |
  |                |  column     | |
  |                |  headers    | |
  |                |             | |
  |                |             | |
  |                |-------------| |
  | |-------------||-------------| |
  | |             ||             | |
  | |             ||             | |
  | | row headers || table body  | |
  | |             ||             | |
  | |             ||             | |
  | |-------------||-------------| |
  |           table view           |
  |--------------------------------|

Note: the row headers subtable cant have css attributes

def method_missing method_sym, *args, &blk
  if Attributes.include? method_sym.to_s.chomp('=')
    set_or_return method_sym.to_s, *args, &blk

  elsif @body.respond_to? method_sym
    @body.send method_sym, *args, &blk

  else
    super
  end
end

def cell col, row, value=nil

    @cell[col][row] ||= {}
  when Hash
    @cell[col][row] ||= {}
    @cell[col][row].merge! value
  else

  end

def data data=nil
      case data
      when Array
        # An array of objects, or an array of arrays, or and array of hashes

        # @data = [[1,"a"],[2,"b"],[3,"c"]]

        @rows = []
        @data = []
        data.collect do |obj|
          # @rows << obj.name
          row = []
          @cols.each do |col|
            row << obj.send(col.to_s.snake_case.to_sym)
          end
          @data << row
        end          

    end

    def cell col, row, value=nil
      case value
      when nil
        @cell[col][row] ||= {}
      when Hash
        @cell[col][row] ||= {}
        @cell[col][row].merge! value
      else
        raise "unsupported data type"
      end
    end


    IgnoredCellAttributes = %w[ data ]

    def set_or_return cell, attribute, value=nil
      case value
      when nil
        cell[attribute]
      else
        validate_value 
      
    end

    def set_or_return key_type, key, value=nil
      # puts "#{method_name}, key_type: #{key_type.inspect}, key: #{key.inspect}, value: #{value.inspect}"
      case value
      when nil
        @orig[key]
      else
        validate_value key_type, key, value unless key_type == nil
        @hash[key] = value
      end
    end

    def cell_attribute col, row, attribute=nil, value=nil
      case attribute
      when Symbol, String, nil
        set_or_return col, row, attribute, value
      else
        raise "unsupported attribute type #{attribute.inspect}. Should be a symbol or string"
      end
    end

def inspect start_col=0
  col0 = start_col + row_header.ascii_col_width
  bw = body.ascii_col_width
  chw = col_header.ascii_col_width
  cha,ba = [0,0]
  if bw > chw
    cha += (bw-chw)/2
  elsif chw > bw
    ba += (chw-bw)/2
  end
  rh = row_header.inspect(start_col).split "\n"
  b = body.inspect(start_col+col0+ba).split "\n"
  (0..([rh.size, b.size].min-1)).each do |i|
    b[i][0,col0] = rh[i][0,col0]
  end
  col_header.inspect(start_col+col0+cha) + b.join("\n") + "\n"
end
