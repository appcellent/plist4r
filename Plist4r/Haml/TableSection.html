def method_missing method_sym, *args, &blk
  if Attributes.include? method_sym.to_s
    value_for method_sym, *args, &blk

  elsif Attributes.include? method_sym.to_s.chomp('=')
    set method_sym, *args, &blk

  elsif @cells.respond_to? method_sym
    @cells.send method_sym, *args, &blk

  else
    super
  end
end
