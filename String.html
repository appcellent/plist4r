def camelcase
  str = self.dup.capitalize.gsub(/[-_.\s]([a-zA-Z0-9])/) { $1.upcase } \
                .gsub('+', 'x')
end

def snake_case
  str = self.dup.gsub(/[A-Z]/) {|s| "_" + s}
  str = str.downcase.sub(/^\_/, "")
end
